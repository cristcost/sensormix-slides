      Tag tag = getIntent().getParcelableExtra(NfcAdapter.EXTRA_TAG);
      StringBuilder id = new StringBuilder();
      byte[] data = tag.getId();
      for (int i = 0; i < data.length; i++) {
        id.append(String.format("%02x", data[i]));
        if (i < data.length - 1) {
          id.append(":");
        }
